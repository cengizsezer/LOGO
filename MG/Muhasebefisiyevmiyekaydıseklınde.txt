select F.JOURNALNO AS [Yevmiye NO],
CASE F.TRCODE WHEN 1 THEN 'Açılış' WHEN 2 THEN 'Tahsil'
WHEN 3 THEN 'Tediye' WHEN 4 THEN 'Mahsup' WHEN 6 THEN 'Kur Farkı'
WHEN 7 THEN 'Kapanış'
END AS [Fiş Türü] , F.FICHENO AS [Fiş No],
convert(varchar, F.DATE_,104) AS 'Fiş Tarihi', L.KEBIRCODE AS [Kebir Kodu],E.CODE AS [Muhasebe Kodu],
E.DEFINITION_ AS [Hesap Açıklaması], L.LINEEXP AS [Açıklama],
L.DEBIT AS [Borç],L.CREDIT AS [Alacak]
FROM LG_010_01_EMFICHE F
INNER JOIN LG_010_01_EMFLINE L ON L.ACCFICHEREF=F.LOGICALREF
INNER JOIN LG_010_EMUHACC E ON E.LOGICALREF=L.ACCOUNTREF
WHERE F.CANCELLED=0
ORDER BY F.JOURNALNO,F.DATE_,L.SIGN
