SELECT

ISNULL(CASE WHEN INV.FROMKASA = 1 AND INV.CANCELLED = 1 THEN ‘İ’ WHEN INV.FROMKASA = 1 THEN ‘K’ END,”) ‘ ‘,
CONVERT(NVARCHAR,INV.DATE_,103) ‘TARİH’,
INV.FICHENO ‘FİŞ NO.’,
INV.DOCODE ‘BELGE NO’,
CASE INV.TRCODE WHEN 8 THEN ‘TOPTAN SATIŞ FATURASI’ WHEN 3 THEN ‘TOPTAN SATIŞ İADE FATURASI’ END ‘TÜRÜ’ ,
INV.NETTOTAL ‘TUTAR’,
CASE INV.TRCURR
WHEN 0 THEN CAST(INV.TRNET AS NVARCHAR)
WHEN 1 THEN CAST(INV.TRNET AS NVARCHAR) + ‘ $’
WHEN 20 THEN CAST(INV.TRNET AS NVARCHAR) + ‘ €’
END ‘DÖVİZLİ TUTAR’,
CPD.NR ‘İŞYERİ’,
CPW.NR ‘AMBAR’

FROM LG_005_01_INVOICE INV
LEFT JOIN LG_005_CLCARD CLC ON CLC.LOGICALREF = INV.CLIENTREF
LEFT JOIN L_CAPIDEPT CPD ON CPD.NR = INV.BRANCH AND CPD.FIRMNR = 5
LEFT JOIN L_CAPIWHOUSE CPW ON CPW.NR = INV.SOURCEINDEX AND CPW.FIRMNR = 5

WHERE GRPCODE = 2

ORDER BY INV.DATE_ ASC
